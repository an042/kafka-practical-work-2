# Шаг 1 — Создание топика через консоль

kafka-topics.sh --create \
  --topic events \
  --partitions 3 \
  --replication-factor 2 \
  --bootstrap-server localhost:9092

# Шаг 2 — Подробная информация о топике

kafka-topics.sh --describe \
  --topic events \
  --bootstrap-server localhost:9092

# Ожидаемый вывод команды --describe:
#
# Topic: events   TopicId: <id>   PartitionCount: 3   ReplicationFactor: 2   Configs: ...
#   Topic: events   Partition: 0   Leader: 1   Replicas: 1,2   Isr: 1,2
#   Topic: events   Partition: 1   Leader: 2   Replicas: 2,3   Isr: 2,3
#   Topic: events   Partition: 2   Leader: 3   Replicas: 3,1   Isr: 3,1
